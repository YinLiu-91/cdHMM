200 template <typename S> static double
201 try_corner_move (const double coeff,
202  const nmsimplex_state_t &state,
203  size_t corner,
204  vector<double> &xc, FuncNd<S> *f)
205 {
206  /* moves a simplex corner scaled by coeff (negative value represents
207  mirroring by the middle point of the "other" corner points)
208  and gives new corner in xc and function value at xc as a
209  return value
210  */
211 
212  const size_t P = state.x1.size();
213  double newval;
214 
215  /* xc = (1-coeff)*(P/(P-1)) * center(all) + ((P*coeff-1)/(P-1))*x_corner */
216  {
217  double alpha = (1 - coeff) * P / (P - 1.0);
218  double beta = (P * coeff - 1.0) / (P - 1.0);
219  vector<double> row = state.x1[corner];
220 
221  xc = (state.center);
222  dscal (alpha, xc);
223  daxpy (beta, row, xc);
224  }
225 
226  newval = f->eval(xc);
227 
228  return newval;
229 }

232 static void
233 update_point (nmsimplex_state_t &state, size_t i,
234  vector<double> &x, double val)
235 {
236  const size_t P = state.x1.size();
237 
238  /* Compute delta = x - x_orig */
239  state.delta = x;
240  daxpy(-1.0, state.x1[i], state.delta);
241 
242  /* Compute xmc = x_orig - c */
243  state.xmc = state.x1[i];
244  daxpy(-1, state.center, state.xmc);
245 
246  /* Update size: S2' = S2 + (2/P) * (x_orig - c).delta + (P-1)*(delta/P)^2 */
247  {
248  double d = dnrm2 (state.delta);
249  double xmcd;
250 
251  xmcd = ddot (state.xmc, state.delta);
252  state.S2 += (2.0 / P) * xmcd + ((P - 1.0) / P) * (d * d / P);
253  }
254 
255  /* Update center: c' = c + (x - x_orig) / P */
256 
257  {
258  double alpha = 1.0 / P;
259  daxpy (-alpha, state.x1[i], state.center);
260  daxpy (alpha, x, state.center);
261  }
262 
263  state.x1[i] = x;
264  state.y1[i] = val;
265 }

267 template <typename S> static int
268 contract_by_best (nmsimplex_state_t &state, size_t best,
269  vector<double> &xc, FuncNd<S> *f)
270 {
271 
272  /* Function contracts the simplex in respect to best valued
273  corner. That is, all corners besides the best corner are moved.
274  (This function is rarely called in practice, since it is the last
275  choice, hence not optimised - BJG) */
276 
277  /* the xc vector is simply work space here */
278 
279 
280  size_t i, j;
281  double newval;
282 
283  int status = MYGSL_SUCCESS;
284 
285  for (i = 0; i < state.x1.size(); i++)
286  {
287  if (i != best)
288  {
289  for (j = 0; j < state.x1[i].size(); j++)
290  {
291  state.x1[i][j] = 0.5 * (state.x1[i][j] + state.x1[best][j]);
292  }
293 
294  /* evaluate function in the new point */
295  state.y1[i] = f->eval( state.x1[i] );
296 
297  /* notify caller that we found at least one bad function value.
298  we finish the contraction (and do not abort) to allow the user
299  to handle the situation */
300 
301  if (!isfinite (newval))
302  {
303  status = MYGSL_EBADFUNC;
304  }
305  }
306  }
307 
308 
309  /* We need to update the centre and size as well */
310  compute_center (state, state.center);
311  compute_size (state, state.center);
312 
313  return status;
314 }

316 static int
317 compute_center (const nmsimplex_state_t &state, vector<double> &center)
318 {
319  /* calculates the center of the simplex and stores in center */
320  const size_t P = state.x1.size();
321  size_t i;
322 
323  set_zero (center);
324 
325  for (i = 0; i < P; i++)
326  {
327  daxpy(1.0, state.x1[i], center);
328  }
329 
330  {
331  const double alpha = 1.0 / P;
332  dscal (alpha, center);
333  }
334 
335  return MYGSL_SUCCESS;
336 }

338 static double
339 compute_size (nmsimplex_state_t &state, vector<double> &center)
340 {
341  /* calculates simplex size as rms sum of length of vectors
342  from simplex center to corner points:
343 
344  sqrt( sum ( || y - y_middlepoint ||^2 ) / n )
345  */
346 
347  vector<vector<double> > x1 = (state.x1);
348  const size_t P = state.x1.size();
349  size_t i;
350 
351  double ss = 0.0;
352 
353  for (i = 0; i < P; i++)
354  {
355  double t;
356  state.ws1 = state.x1[i];
357  daxpy(-1.0, center, state.ws1);
358  t = dnrm2(state.ws1);
359  ss += t * t;
360  }
361 
362  /* Store squared size in the state */
363  state.S2 = (ss / P);
364 
365  return sqrt (ss / P);
366 }

395 template <typename S> int
396 gsl_multimin_fminimizer_type<S>::set (nmsimplex_state_t &state, FuncNd<S> *f,
397  vector<double> &x,
398  double *size, vector<double> &step_size)
399 {
400  int status;
401  size_t i;
402  double val;
403 
404 
405  //nmsimplex_state_t *state = (nmsimplex_state_t *) vstate;
406 
407  vector<double> xtemp = state.ws1;
408  if (xtemp.size() != x.size())
409  {
410  cerr << "incompatible size of x" << endl; exit(1);
411  }
412 
413  if (xtemp.size() != step_size.size() )
414  {
415  cerr << "incompatible size of step_size" << endl; exit(1);
416  }
417 
418  /* first point is the original x0 */
419  val = f->eval( x );
420 
421  if (!isfinite (val))
422  {
423  cerr << "non-finite function value encountered" << endl; exit(1);
424  }
425 
426  state.x1[0] = x;
427  state.y1[0] = val;
428 
429  /* following points are initialized to x0 + step_size */
430 
431  for (i = 0; i < x.size(); i++)
432  {
433  xtemp = x;
434 
435  {
436  double xi = x[i];
437  double si = step_size[i];
438 
439  xtemp[i] = xi + si;
440  val = f->eval(xtemp);
441  }
442 
443  if (!isfinite (val))
444  {
445  cerr << "non-finite function value encountered" << endl; exit(1);
446  }
447  state.x1[i+1] = xtemp;
448  state.y1[i+1] = val;
449  }
450 
451  compute_center (state, state.center);
452 
453  /* Initialize simplex size */
454  *size = compute_size (state, state.center);
455 
456  state.count++;
457 
458  return MYGSL_SUCCESS;
459 }

461 template <typename S> int
462 gsl_multimin_fminimizer_type<S>::iterate (nmsimplex_state_t &state, FuncNd<S> *f,
463  vector<double> &x, double *size, double *fval)
464 {
465 
466  /* Simplex iteration tries to minimize function f value */
467  /* Includes corrections from Ivo Alxneit <ivo.alxneit@psi.ch> */
468 
469  //nmsimplex_state_t *state = (nmsimplex_state_t *) vstate;
470 
471  /* xc and xc2 vectors store tried corner point coordinates */
472  vector<double> xc = state.ws1;
473  vector<double> xc2 = state.ws2;
474 
475 
476  const size_t n = state.y1.size();
477  size_t i;
478  size_t hi, s_hi, lo;
479  double dhi, ds_hi, dlo;
480  int status;
481  double val, val2;
482 
483  if (xc.size() != x.size() )
484  {
485  cerr << "incompatible size of x" << endl; exit(1);
486  }
487 
488  /* get index of highest, second highest and lowest point */
489 
490  dhi = dlo = state.y1[0];
491  hi = 0;
492  lo = 0;
493 
494  ds_hi = state.y1[1];
495  s_hi = 1;
496 
497  for (i = 1; i < n; i++)
498  {
499  val = state.y1[i];
500  if (val < dlo)
501  {
502  dlo = val;
503  lo = i;
504  }
505  else if (val > dhi)
506  {
507  ds_hi = dhi;
508  s_hi = hi;
509  dhi = val;
510  hi = i;
511  }
512  else if (val > ds_hi)
513  {
514  ds_hi = val;
515  s_hi = i;
516  }
517  }
518 
519  /* try reflecting the highest value point */
520  val = try_corner_move (-1.0, state, hi, xc, f);
521 
522  if (isfinite (val) && val < state.y1[lo])
523  {
524  /* reflected point is lowest, try expansion */
525  val2 = try_corner_move (-2.0, state, hi, xc2, f);
526 
527  if (isfinite (val2) && val2 < state.y1[lo])
528  {
529  update_point (state, hi, xc2, val2);
530  }
531  else
532  {
533  update_point (state, hi, xc, val);
534  }
535  }
536  else if (!isfinite (val) || val > state.y1[s_hi])
537  {
538  /* reflection does not improve things enough, or we got a
539  non-finite function value */
540 
541  if (isfinite (val) && val <= state.y1[hi])
542  {
543  /* if trial point is better than highest point, replace
544  highest point */
545  update_point (state, hi, xc, val);
546  }
547 
548  /* try one-dimensional contraction */
549  val2 = try_corner_move (0.5, state, hi, xc2, f);
550 
551  if (isfinite (val2) && val2 <= state.y1[hi])
552  {
553  update_point (state, hi, xc2, val2);
554  }
555  else
556  {
557  /* contract the whole simplex about the best point */
558  status = contract_by_best (state, lo, xc, f);
559 
560  if (status != MYGSL_SUCCESS)
561  {
562  cerr << "contraction failed" << endl; exit(1);
563  }
564  }
565  }
566  else
567  {
568  /* trial point is better than second highest point. Replace
569  highest point by it */
570  update_point (state, hi, xc, val);
571  }
572 
573  /* return lowest point of simplex as x */
574  lo = min_index(state.y1);
575  x = state.x1[lo];
576 
577  *fval = state.y1[lo];
578 
579  /* Update simplex size */
580  {
581  double S2 = state.S2;
582 
583  if (S2 > 0)
584  {
585  *size = sqrt (S2);
586  }
587  else
588  {
589  /* recompute if accumulated error has made size invalid */
590  *size = compute_size (state, state.center);
591  }
592  }
593 
594  return MYGSL_SUCCESS;
595 }
